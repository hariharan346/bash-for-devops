01 / INTRODUCTION

What is Bash?

What is a Shell?

A shell is a command interpreter — a program that reads commands from a user or a script, and executes them by communicating with the operating system kernel. The shell acts as the bridge between human-readable instructions and low-level system calls.

What is Bash?

Bash (Bourne Again SHell) is the GNU Project's replacement for the original Bourne Shell (sh). Created by Brian Fox in 1989, Bash is the default login shell on most Linux distributions and macOS (pre-Catalina). It is fully POSIX-compliant and adds significant improvements: command-line editing, job control, shell functions, arrays, and more.

ℹ
Version: This reference covers Bash 5.2+ features. Check your version with bash --version.

02 / FUNDAMENTALS

Bash Basics

Shell Syntax

A Bash script is a sequence of commands, each on its own line or separated by ;. Bash reads commands left to right, top to bottom. Blank lines and lines beginning with # are ignored.

The Shebang

The first line of every script should be a shebang — it tells the OS which interpreter to use.

bash shebang.sh 
#!/usr/bin/env bash     # Preferred: uses $PATH to find bash
#!/bin/bash             # Hardcoded path — less portable
#!/bin/sh               # POSIX sh — more portable, fewer features
💡
DevOps Tip: Always use #!/usr/bin/env bash. It works on any system where Bash is installed, regardless of the exact path. This is critical for portability across Alpine Linux, macOS, and Ubuntu containers.

Script Structure

bash script_structure.sh 
#!/usr/bin/env bash
# ─────────────────────────────────────────
# Script: deploy.sh
# Description: Deploy application to server
# Author: DevOps Team
# Date: 2025-01-15
# ─────────────────────────────────────────

# 1. Strict mode
set -euo pipefail

# 2. Constants
readonly APP_NAME="myapp"
readonly LOG_FILE="/var/log/deploy.log"

# 3. Functions
log() {
  echo "[$(date +'%Y-%m-%d %H:%M:%S')] $*" | tee -a "$LOG_FILE"
}

main() {
  log "Starting deployment of $APP_NAME"
  # ... deployment logic ...
  log "Deployment complete"
}

# 4. Entry point
main "$@"

Running Scripts

bash execution methods 
# Make executable and run directly
chmod +x script.sh
./script.sh

# Run with bash explicitly
bash script.sh

# Source — runs in current shell (shares env)
source script.sh
. script.sh          # POSIX equivalent

# Pass arguments
./deploy.sh production v2.1.0

# Debug mode
bash -x script.sh

Basic Commands Syntax

bash command syntax

# Multiple commands on one line

# Subshell grouping

# Long command continuation

  --name myapp \
  -p 8080:80 \

Variable Manipulation

bash var_manipulation.sh 
FILE="/var/log/nginx/access.log.gz"

# String length
echo ${#FILE}                    # 30

# Substring
echo ${FILE:9:5}                 # nginx

# Remove prefix (shortest match)
echo ${FILE#*/}                  # var/log/nginx/access.log.gz

# Remove prefix (longest match)
echo ${FILE##*/}                 # access.log.gz

# Remove suffix (shortest match)

# Remove suffix (longest match)
echo ${FILE%%.*}                 # /var/log/nginx/access

# Replace first match

# Replace all matches
echo ${FILE//log/LOG}            # /var/LOG/nginx/access.LOG.gz

# Uppercase / lowercase
ENV="production"
echo ${ENV^^}                    # PRODUCTION
echo ${ENV^}                     # Production
echo ${ENV,,}                    # production

Arithmetic Operators

bash arithmetic.sh 
# (( )) — arithmetic context
(( RESULT = 10 + 5 ))
(( COUNT++ ))
(( TOTAL += 100 ))

Logical Operators

bash logical.sh 
# AND — run second only if first succeeds
mkdir -p /opt/app && cp app.tar.gz /opt/app/

systemctl start nginx || echo "nginx failed to start"

if [ $CPU -gt 80 -a $MEM -gt 90 ]; then
  echo "Alert"
fi

07 / LOOPS

Loops

for Loop

bash for_loops.sh 
# Iterate over list
for server in web1 web2 web3 web4; do
  echo "Deploying to $server..."
  ssh "$server" "sudo systemctl restart app"
done

# C-style for loop
for (( i=1; i<=5; i++ )); do
  echo "Attempt $i/5"
  curl -sf http://api/health && break
  sleep 5
done

# Range with brace expansion
for port in {8080..8084}; do
  nc -z localhost "$port" && echo "Port $port open"
done

# Iterate files
for config in /etc/nginx/conf.d/*.conf; do
  echo "Checking $config"
  nginx -t -c "$config"
done

while Loop

bash while_loop.sh 
# Wait for service to be ready
MAX_WAIT=60
ELAPSED=0
while ! curl -sf http://localhost:8080/health; do
  if (( ELAPSED >= MAX_WAIT )); then
    echo "Service failed to start in ${MAX_WAIT}s"
    exit 1
  fi
  sleep 2; (( ELAPSED += 2 ))
done

# Infinite loop — monitoring daemon
while true; do
  check_disk_usage
  check_memory
  sleep 60
done

# Read file line by line
while IFS=',' read -r host port service; do
  check_service "$host" "$port" "$service"
done < services.csv

until Loop

bash until.sh 
# until runs as long as condition is FALSE
RETRIES=0
until pg_isready -h localhost; do
  (( RETRIES++ ))
  [[ $RETRIES -ge 10 ]] && { echo "DB never ready"; exit 1; }
  sleep 3
done

Loop Control: break and continue

bash break_continue.sh 
for server in "${SERVERS[@]}"; do
  # Skip offline servers
  if ! ping -c1 -W2 "$server" >/dev/null 2&1; then
    echo "Skipping offline: $server"
    continue
  fi

  # Stop if deploy fails
  if ! deploy_to "$server"; then
    echo "Deploy failed on $server. Halting."
    break
  fi
done

# break/continue with nested loops
for region in us eu ap; do
  for zone in 1 2 3; do
    [[ "$region" == eu && $zone -eq 3 ]] && continue 2
    echo "$region-$zone"
  done
done

08 / FUNCTIONS

Functions

Creating Functions

bash functions.sh 
# Two equivalent syntaxes
greet() { echo "Hello, $1"; }

function greet {
  echo "Hello, $1"
}

# Call a function
greet "DevOps"

# Local variables (don't leak)
check_port() {
  local host="$1"
  local port="$2"
  local timeout="${3:-5}"  # default 5s
  nc -z -w "$timeout" "$host" "$port"
}

check_port db.prod.local 5432 10

Return Values

bash return_values.sh 
# return sets exit code only (0-255)
is_healthy() {
  curl -sf "http://$1/health" >/dev/null
  return $?
}

if is_healthy "api.prod.local"; then
  echo "API is UP"
fi

# Return string via echo + command substitution
get_version() {
  local app="$1"
  docker inspect --format '{{.Config.Labels.version}}' "$app"
}

VER=$(get_version myapp)
echo "Running version: $VER"

# Return via nameref (Bash 4.3+)
get_disk_usage() {
  local -n __result="$1"
  __result=$(df -h / | awk 'NR==2{print $5}')
}
get_disk_usage DISK_PCT
echo "Disk: $DISK_PCT"
💡
Best Practice: Define a main() function and call it at the end of every script. This makes scripts testable (you can source them without executing), and the structure is immediately clear to teammates.

Arrays

Indexed Arrays

bash indexed_arrays.sh 
# Declare and populate
SERVERS=(web1 web2 web3 web4)
PORTS=(80 443 8080 8443)

declare -a SERVICES        # explicit declaration
SERVICES+=(nginx)           # append
SERVICES+=(mysql redis)

# Delete array
unset SERVERS

Associative Arrays (Bash 4+)

bash assoc_arrays.sh 
# Must declare explicitly
declare -A SERVICE_PORTS
SERVICE_PORTS[nginx]=80
SERVICE_PORTS[mysql]=3306
SERVICE_PORTS[redis]=6379
SERVICE_PORTS[postgres]=5432

# Iterate key-value pairs
for svc in "${!SERVICE_PORTS[@]}"; do

    echo "✓ $svc:${SERVICE_PORTS[$svc]}"
done

10 / EXPANSIONS

Expansions

All Expansion Types

bash expansions.sh 
# 1. Variable Expansion
echo "$HOME"
echo "${USER}"
echo "${PATH:-/usr/bin}"     # default if unset
echo "${#HOSTNAME}"          # string length

# 2. Brace Expansion
mkdir -p /opt/app/{logs,config,data,tmp}
cp nginx.conf{,.bak}       # backup a file
echo {1..5}                # 1 2 3 4 5
echo {a..e}                # a b c d e
echo {web,db,cache}{1,2}   # web1 web2 db1 db2 cache1 cache2

# 3. Command Substitution
DATE=$(date +%Y%m%d)
GIT_SHA=$(git rev-parse --short HEAD)
IMAGE_TAG="$APP:$GIT_SHA-$DATE"

# 4. Arithmetic Expansion
echo $(( 1024 * 1024 ))      # 1048576
TIMEOUT=$(( $RETRIES * 5 ))

# 5. Filename (Glob) Expansion
for f in /var/log/*.log; do gzip "$f"; done
for f in **/*.yaml; do kubectl apply -f "$f"; done

# 6. Process Substitution
diff <(ssh server1 "cat /etc/hosts") <(ssh server2 "cat /etc/hosts")

# 7. Here String
grep "ERROR" <<< "$LOG_LINE"
⚠
Quote your expansions! Always double-quote variable expansions: "$VAR" not $VAR. Unquoted variables split on whitespace and undergo glob expansion, causing subtle bugs in filenames with spaces.

11 / REDIRECTION

Redirection

File Descriptors

Every process has three default file descriptors: 0 (stdin), 1 (stdout), 2 (stderr).

bash redirection.sh 
# stdout to file (overwrite)
ls -la > file_list.txt

# stdout append
echo "$LOG_ENTRY" >> /var/log/app.log

# stderr to file
nginx -t 2> nginx-errors.txt

# stderr to stdout (merge)
./deploy.sh 2&1 | tee deploy.log

# Both stdout and stderr to file
make build > build.log 2&1
make build &> build.log   # bash shorthand

# Discard output
systemctl is-active nginx >/dev/null 2&1

# stdin from file
mysql -u root < schema.sql

# Here document
cat <<EOF > /etc/app/config.yaml
server:
  host: $DB_HOST
  port: $DB_PORT
  name: $DB_NAME
EOF

# Here doc no variable expansion
cat <<'EOF'
  This $VARIABLE is not expanded
EOF

# Named pipe (FIFO)
mkfifo mypipe
gzip -c < mypipe > archive.gz &
tar -c /data > mypipe
ℹ
DevOps Pattern: Always redirect both stdout and stderr to your log files in CI/CD scripts: ./deploy.sh >> deploy.log 2>&1. This ensures nothing is silently lost during automated runs.

12 / PROCESS MANAGEMENT

Process Management

bash processes.sh 
# Background process
long_task.sh &
BG_PID=$!

# Parallel deployments
for server in "${SERVERS[@]}"; do
  deploy_to "$server" &
done
wait   # wait for ALL background jobs
echo "All deployments done"

# wait with specific PID + exit code
task_a &; PID_A=$!
task_b &; PID_B=$!
wait $PID_A || echo "task_a failed"
wait $PID_B || echo "task_b failed"

# Job control
jobs          # list background jobs
fg %1         # bring job 1 to foreground
bg %2         # resume job 2 in background
kill %1       # kill job 1

# Process substitution for parallel
diff <(sort file1) <(sort file2)

# nohup — immune to hangup signal
nohup ./monitor.sh > monitor.log 2&1 &

# Limit parallel jobs to N
MAX_JOBS=4
for item in "${ITEMS[@]}"; do
  while [[ $(jobs -r | wc -l) -ge $MAX_JOBS ]]; do
    sleep 0.5
  done
  process_item "$item" &
done
wait

13 / ERROR HANDLING

Error Handling

Strict Mode

bash strict_mode.sh 
#!/usr/bin/env bash
set -e          # exit on any error
set -u          # error on undefined variables
set -o pipefail # catch errors in pipes
set -x          # debug: print each command (remove in prod)

# All at once
set -euo pipefail

# Temporarily disable -e for expected failures
set +e
grep "pattern" file.txt   # may return 1 if no match
GREP_EXIT=$?
set -e

trap — Signal Handling

bash trap.sh 
# Cleanup on exit (success or failure)
TMPDIR=$(mktemp -d)

cleanup() {
  echo "Cleaning up $TMPDIR"
  rm -rf "$TMPDIR"
}

trap cleanup EXIT          # runs on any exit
trap cleanup INT TERM      # Ctrl-C or kill

# Error trap with line number
error_handler() {
  echo "ERROR at line $1: command failed" >&2
  echo "Stack trace:" >&2
  local i
  for (( i=1; i<${#FUNCNAME[@]}; i++ )); do
    echo "  [$i] ${FUNCNAME[$i]}() at line ${BASH_LINENO[$i-1]}" >&2
  done
}

trap 'error_handler $LINENO' ERR

# Graceful shutdown for long-running processes
handle_sigterm() {
  echo "Received SIGTERM, shutting down gracefully..."
  # finish current work...
  exit 0
}
trap handle_sigterm TERM

getopts — Argument Parsing

bash getopts.sh 
#!/usr/bin/env bash
# Usage: ./deploy.sh -e production -v 2.1.0 -d

ENVIRONMENT="staging"
VERSION="latest"
DRY_RUN=false

while getopts "e:v:dh" opt; do
  case $opt in
    e) ENVIRONMENT="$OPTARG" ;;
    v) VERSION="$OPTARG" ;;
    d) DRY_RUN=true ;;
    h) usage; exit 0 ;;
    ?) echo "Unknown option" >&2; exit 2 ;;
  esac
done

echo "Deploying $VERSION to $ENVIRONMENT (dry-run: $DRY_RUN)"

# View crontab

# Crontab format:
# ┌─ minute (0-59)
# │ ┌─ hour (0-23)

    deployment/"$DEPLOYMENT" --timeout=5m; then

  kubectl -n "$NAMESPACE" rollout undo deployment/"$DEPLOYMENT"
  exit 1
fi

# Get failing pods
kubectl -n "$NAMESPACE" get pods \
  --field-selector status.phase=Failed \
  -o 'jsonpath={range .items[*]}{.metadata.name}{"\n"}{end}'

# Scale deployment
kubectl scale deployment/"$DEPLOYMENT" --replicas=5 -n "$NAMESPACE"

# Exec into pod
POD=$(kubectl -n "$NAMESPACE" get pods \
  -l app="$DEPLOYMENT" -o 'jsonpath={.items[0].metadata.name}')
kubectl -n "$NAMESPACE" exec -it "$POD" -- bash

1. Server Health Monitor

bash server_health.sh 
#!/usr/bin/env bash
# ─────────────────────────────────────────
# Server Health Monitor
# Usage: ./server_health.sh [--slack-webhook URL]
# ─────────────────────────────────────────
set -euo pipefail

# Config
SLACK_WEBHOOK="${SLACK_WEBHOOK:-}"
CPU_THRESHOLD=80
MEM_THRESHOLD=85
DISK_THRESHOLD=90
HOSTNAME=$(hostname -f)
REPORT=""
ALERTS=0

log() { echo "[$(date '+%Y-%m-%d %H:%M:%S')] $*"; }
alert() { REPORT+="⚠ $*\n"; (( ALERTS++ )); log "ALERT: $*"; }

check_cpu() {
  local cpu
  cpu=$(top -bn1 | grep "Cpu(s)" | awk '{print int($2+$4)}')
  log "CPU: ${cpu}%"
  (( cpu > CPU_THRESHOLD )) && alert "CPU at ${cpu}% (threshold: $CPU_THRESHOLD%)"
}

check_memory() {
  local used total pct
  read -r total used <<< $(free -m | awk '/^Mem:/{print $2, $3}')
  pct=$(( used * 100 / total ))
  log "Memory: ${used}MB / ${total}MB (${pct}%)"
  (( pct > MEM_THRESHOLD )) && alert "Memory at ${pct}% (${used}MB/${total}MB)"
}

check_disk() {
  while IFS= read -r line; do
    local pct mount
    pct=$(echo "$line" | awk '{print int($5)}')
    mount=$(echo "$line" | awk '{print $6}')
    log "Disk $mount: ${pct}%"
    (( pct > DISK_THRESHOLD )) && alert "Disk $mount at ${pct}%"
  done < <(df -h | grep -v 'tmpfs\|udev\|Filesystem')
}

check_services() {
  local -a svcs=(nginx mysql docker)
  for svc in "${svcs[@]}"; do
    if ! systemctl is-active --quiet "$svc" 2>/dev/null; then
      alert "Service $svc is NOT running"
    else
      log "Service $svc: OK"
    fi
  done
}

send_slack() {
  [[ -z "$SLACK_WEBHOOK" ]] || [[ $ALERTS -eq 0 ]] && return
  curl -sf -X POST -H 'Content-type: application/json' \
    --data "{\"text\":\"🚨 *$HOSTNAME* Health Alerts:\n$REPORT\"}" \
    "$SLACK_WEBHOOK"
}

main() {
  log "=== Health Check: $HOSTNAME ==="
  check_cpu
  check_memory
  check_disk
  check_services
  send_slack
  log "Check complete. Alerts: $ALERTS"
  [[ $ALERTS -eq 0 ]] || exit 1
}

main

2. Auto Deployment Script

bash auto_deploy.sh 
#!/usr/bin/env bash
# Zero-downtime deploy with rollback
set -euo pipefail

APP="${APP_NAME:?APP_NAME required}"
VERSION="${VERSION:?VERSION required}"
DEPLOY_DIR="/opt/$APP"
RELEASES_DIR="$DEPLOY_DIR/releases"
CURRENT_LINK="$DEPLOY_DIR/current"
RELEASE_PATH="$RELEASES_DIR/$VERSION"
MAX_RELEASES=5

log()      { echo "[$(date +%T)] ▶ $*"; }
success()  { echo "[$(date +%T)] ✓ $*"; }
fail()     { echo "[$(date +%T)] ✗ $*" >&2; exit 1; }

rollback() {
  log "Rolling back to previous release..."
  local prev
  prev=$(ls -1dt "$RELEASES_DIR"/* | sed -n '2p')
  [[ -d "$prev" ]] || fail "No previous release found"
  ln -sfn "$prev" "$CURRENT_LINK"
  systemctl restart "$APP"
  success "Rolled back to $(basename $prev)"
}

trap rollback ERR

main() {
  log "Deploying $APP v$VERSION"

  # Prepare release directory
  mkdir -p "$RELEASE_PATH"
  tar -xzf "$APP-$VERSION.tar.gz" -C "$RELEASE_PATH"
  success "Unpacked release"

  # Symlink shared directories
  ln -sf "$DEPLOY_DIR/shared/config" "$RELEASE_PATH/config"
  ln -sf "$DEPLOY_DIR/shared/logs" "$RELEASE_PATH/logs"

  # Pre-deployment health check
  log "Running pre-deploy tests..."
  bash "$RELEASE_PATH/scripts/pre-deploy.sh"

  # Switch symlink (atomic-ish)
  ln -sfn "$RELEASE_PATH" "$CURRENT_LINK"
  systemctl restart "$APP"

  # Post-deploy health check
  sleep 3
  curl -sf "http://localhost/health" || fail "Post-deploy health check failed"

  # Clean old releases (keep MAX_RELEASES)
  ls -1dt "$RELEASES_DIR"/* | tail -n +$(( MAX_RELEASES + 1 )) | \
    xargs -r rm -rf

  success "Deployed $APP v$VERSION successfully"
}

main

3. Backup Script

bash backup.sh 
#!/usr/bin/env bash
# Database + files backup with S3 upload
set -euo pipefail

DB_NAME="${DB_NAME:?required}"
DB_HOST="${DB_HOST:-localhost}"
S3_BUCKET="${S3_BUCKET:?required}"
BACKUP_DIR="/tmp/backups"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
RETENTION_DAYS=30

log() { echo "[$(date +%T)] $*"; }

trap 'rm -rf "$BACKUP_DIR/$TIMESTAMP"' EXIT

backup_database() {
  log "Dumping database $DB_NAME..."
  local dump_file="$BACKUP_DIR/$TIMESTAMP/$DB_NAME.sql.gz"
  mkdir -p "$(dirname "$dump_file")"
  pg_dump -h "$DB_HOST" -U postgres "$DB_NAME" | \
    gzip -9 > "$dump_file"
  log "DB dump: $(du -sh "$dump_file" | cut -f1)"
  echo "$dump_file"
}

backup_files() {
  log "Archiving application files..."
  local archive="$BACKUP_DIR/$TIMESTAMP/app_files.tar.gz"
  tar -czf "$archive" \
    --exclude='*.log' \
    --exclude='node_modules' \
    /opt/app/uploads /opt/app/config
  echo "$archive"
}

upload_s3() {
  local file="$1"
  local s3_key="backups/$TIMESTAMP/$(basename "$file")"
  log "Uploading to s3://$S3_BUCKET/$s3_key"
  aws s3 cp "$file" "s3://$S3_BUCKET/$s3_key" \
    --storage-class STANDARD_IA
}

purge_old_backups() {
  log "Purging backups older than $RETENTION_DAYS days..."
  aws s3 ls "s3://$S3_BUCKET/backups/" | awk '{print $2}' | \
    while read -r prefix; do
      local date=$(echo "$prefix" | cut -c1-8)
      local age=$(( ( $(date +%s) - $(date -d "$date" +%s) ) / 86400 ))
      (( age > RETENTION_DAYS )) && \
        aws s3 rm --recursive "s3://$S3_BUCKET/backups/$prefix"
    done
}

main() {
  log "=== Backup started: $TIMESTAMP ==="
  DB_DUMP=$(backup_database)
  FILES=$(backup_files)
  upload_s3 "$DB_DUMP"
  upload_s3 "$FILES"
  purge_old_backups
  log "=== Backup complete ==="
}

main

    sort -rh | head -5

  local subject="$1"
  local body="$2"

}

  local usage=${pct%\%}    # strip % sign

      "[DISK] $HOST:$mount at ${usage}%" \
      "Host: $HOST\nMount: $mount\nUsage: ${usage}% ($used of $size)\n\nTop consumers:\n$hogs"

    fi
    echo
  done

6. Service Health Checker

bash service_checker.sh 
#!/usr/bin/env bash
# Multi-endpoint health checker with retry logic
set -euo pipefail

declare -A ENDPOINTS=(
  [API]="http://api.prod.local/health"
  [Auth]="http://auth.prod.local/ping"
  [DB]="http://db-proxy.prod.local/status"
  [Cache]="http://redis.prod.local/health"
)

TIMEOUT=5
RETRIES=3
RETRY_DELAY=2
FAILED_SERVICES=()

check_endpoint() {
  local name="$1" url="$2"
  local attempt=1

  while (( attempt <= RETRIES )); do
    local code
    code=$(curl -s -o /dev/null -w "%{http_code}" \
      --max-time "$TIMEOUT" "$url" 2>/dev/null || echo "000")

    if [[ "$code" == 2* ]]; then
      printf "  \e[32m✓\e[0m %-10s %s [HTTP $code]\n" "$name" "$url"
      return 0
    fi

    (( attempt++ ))
    sleep "$RETRY_DELAY"
  done

  printf "  \e[31m✗\e[0m %-10s %s [HTTP $code after $RETRIES tries]\n" \
    "$name" "$url"
  FAILED_SERVICES+=("$name")
}

echo "Service Health Check — $(date)"
echo "─────────────────────────────────────────"

# Run checks in parallel
for svc in "${!ENDPOINTS[@]}"; do
  check_endpoint "$svc" "${ENDPOINTS[$svc]}" &
done
wait

echo
if (( ${#FAILED_SERVICES[@]} > 0 )); then
  echo "FAILED: ${FAILED_SERVICES[*]}"
  exit 1
fi
echo "All services healthy ✓"
